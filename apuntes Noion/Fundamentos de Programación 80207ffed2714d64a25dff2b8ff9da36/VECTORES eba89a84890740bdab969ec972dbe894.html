Suma vector

def sumaVector(V):
    n = V[0] + 1
    s = 0
    for i in range(1, n):
         s = s + V[i]
    return s

def ordenaAscen(V)

	  k = i
    for j in range(i+1, V[0] + 1):
	    if V[j] < V[k]:
	      k = j
    intercambiar(V, i, k)
